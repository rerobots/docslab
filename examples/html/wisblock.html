#include <Arduino.h>
#include <SPI.h>


void setup()
{
    time_t timeout = millis();
    Serial.begin(115200);
    while (!Serial) {
        if ((millis() - timeout) < 5000) {
            delay(100);
        } else {
            break;
        }
    }

    Serial.println("Bienvenido!");
}


void loop()
{
    delay(1000);
    Serial.println(".");
}
